94 /*
95  DDG extensions include:
96  Each reachability object now has a copy of the key used to store it in a
97  dictionary. This allows each observer to quickly determine if the event is
98  important to them.
99 
100  -currentReachabilityStatus also has a significantly different decision criteria than
101  Apple's code.

103  A multiple convenience test methods have been added.
104  */

110 #define USE_DDG_EXTENSIONS 1 // Use DDG's Extensions to test network criteria.

114 enum {
115 
116  // DDG NetworkStatus Constant Names.
117  eNotReachable = 0, // Apple's code depends upon 'NotReachable' being the same value as 'NO'.
118  eReachableViaWWAN, // Switched order from Apple's enum. WWAN is active before WiFi.
119  eReachableViaWiFi,
120  eReachableVia2G,
121  eReachableVia3G,
122  eReachableVia4G
123 };

126 enum {
127 
128  // Apple NetworkStatus Constant Names.
129  NotEMReachable = eNotReachable,
130  EMReachableViaWiFi = eReachableViaWiFi,
131  EMReachableViaWWAN = eReachableViaWWAN
132 
133 };

151 // Designated Initializer.

154 // Use to check the reachability of a particular host name.

165 + (EMReachability *) reachabilityForLocalWiFi;

167 //Start listening for reachability notifications on the current run loop.

171 // Comparison routines to enable choosing actions in a notification.
172 - (BOOL) isEqual: (EMReachability *) r;
173 
174 // These are the status tests.
175 - (NetworkStatus) currentReachabilityStatus;
176 
177 // The main direct test of reachability.
178 - (BOOL) isReachable;
179 
180 // WWAN may be available, but not active until a connection has been established.
