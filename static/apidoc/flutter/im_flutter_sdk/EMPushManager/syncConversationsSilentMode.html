Future<void> syncConversationsSilentMode() async {
  try {
    Map result = await Client.instance.pushManager
        .callNativeMethod(ChatMethodKeys.syncSilentModels);
    EMError.hasErrorFromResult(result);
  } catch (e) {
    rethrow;
  }
}
