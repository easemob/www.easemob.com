    return DownloadStatus.DOWNLOADING;
  } else if (status == 1) {
    return DownloadStatus.SUCCESS;
  } else if (status == 2) {
    return DownloadStatus.FAILED;
  } else {
    return DownloadStatus.PENDING;
  }
